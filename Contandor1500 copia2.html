<!DOCTYPE html>

<html lang="en">

<head>

    <meta charset="UTF-8">

    <title>Contador Automático</title>

</head>

<body>

    <h1 id="contador">150000</h1>


    <script>

        let cuenta = 150000;


        function incrementarContador() {

            cuenta++;

            document.getElementById("contador").innerText = cuenta.toLocaleString("en-US");

        }


        // Actualizar el contador cada 10 milisegundos

        setInterval(incrementarContador, 10);

    </script>

</body>

</html>
